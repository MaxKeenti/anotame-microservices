Create new file src/lib/services/ApiError.ts with custom error class exposing HTTP status code:

```typescript
export class ApiError extends Error {
  constructor(
    message: string,
    public status: number,
    public body?: unknown
  ) {
    super(message);
    this.name = 'ApiError';
  }
}

export function isApiError(error: unknown): error is ApiError {
  return error instanceof ApiError;
}
```

This provides type-safe error handling for all API errors with status code accessible to catch blocks.
