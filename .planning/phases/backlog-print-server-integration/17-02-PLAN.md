**Change 2 — Add handleBulkPrint() function** after `handleBulkCancel()` (around line 151).
The function must:
- Cap at 10 orders with a warning toast if exceeded (per D-11)
- Show a progress toast before opening windows (per D-10 / CONTEXT.md specifics)
- Open customer ticket then work order tag per order, 600ms between orders (per D-10)
- Use establishment name "ANOTAME" as fallback (no establishment state on this page)

```typescript
function handleBulkPrint() {
  const MAX_PRINT = 10;
  const orders = selectedOrders;

  if (orders.length === 0) return;

  if (orders.length > MAX_PRINT) {
    toast.warning(`Solo se imprimirán los primeros ${MAX_PRINT} pedidos de ${orders.length} seleccionados.`);
  }

  const batch = orders.slice(0, MAX_PRINT);
  toast.info(`Imprimiendo ${batch.length} pedido${batch.length !== 1 ? 's' : ''}...`);

  batch.forEach((order, index) => {
    setTimeout(() => {
      // Customer ticket
      const receiptHtml = generateReceiptHtml({
        ticketNumber: order.ticketNumber,
        customerName: `${order.customer.firstName} ${order.customer.lastName}`,
        phone: order.customer.phoneNumber,
        deadline: order.committedDeadline || new Date().toISOString(),
        pickupCode: order.pickupCode || undefined,
        items: order.items.map((i: any) => ({
          garment: i.garmentName,
          services: i.services?.map((s: any) => ({
            name: s.serviceName,
            price: s.unitPrice,
            adjustment: s.adjustmentAmount || undefined,
            adjustmentReason: s.adjustmentReason || undefined,
          })) || [],
          notes: i.notes || undefined,
        })),
        total: order.totalAmount,
        amountPaid: order.amountPaid || 0,
        balance: Math.max(0, (order.totalAmount || 0) - (order.amountPaid || 0)),
        establishment: { name: 'ANOTAME' },
      });

      const ticketWin = window.open('', '_blank', 'width=400,height=600');
      if (ticketWin) {
        ticketWin.document.write(receiptHtml);
        ticketWin.document.close();
        ticketWin.setTimeout(() => { ticketWin.focus(); ticketWin.print(); ticketWin.close(); }, 250);
      }

      // Work order tag — 300ms after the ticket window
      setTimeout(() => {
        const workOrderHtml = generateWorkOrderHtml({
          ticketNumber: order.ticketNumber,
          customerName: `${order.customer.firstName} ${order.customer.lastName}`,
          phone: order.customer.phoneNumber,
          deadline: order.committedDeadline || new Date().toISOString(),
          notes: order.notes || undefined,
          items: order.items.map((i: any) => ({
            garment: i.garmentName,
            services: i.services?.map((s: any) => ({
              name: s.serviceName,
              adjustmentReason: s.adjustmentReason || undefined,
            })) || [],
            notes: i.notes || undefined,
          })),
        });

        const woWin = window.open('', '_blank', 'width=500,height=700');
        if (woWin) {
          woWin.document.write(workOrderHtml);
          woWin.document.close();
          woWin.setTimeout(() => { woWin.focus(); woWin.print(); woWin.close(); }, 250);
        }
      }, 300);

    }, index * 600);  // 600ms between each order to avoid popup blocker
  });
}
```
